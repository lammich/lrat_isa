Theory Intf_Dfun

theory Intf_Dfun
imports Isabelle_LLVM.IICF
begin

text ‹Functions with default value›

definition [to_relAPP]: "dfun_rel K V ≡ K→V"
sepref_decl_intf ('k,'v) i_dfun is "'k ⇒ 'v"

lemma [synth_rules]: "⟦INTF_OF_REL K TYPE('k); INTF_OF_REL V TYPE('v)⟧ 
  ⟹ INTF_OF_REL (⟨K,V⟩dfun_rel) TYPE(('k,'v) i_dfun)" by simp
    
context
  fixes dflt :: 'a
begin 

  definition [simp]: "op_dfun_empty = (λ_. dflt)"
  definition [simp]: "op_dfun_lookup f k = f k"
  definition [simp]: "op_dfun_upd f k v = f(k:=v)"
  definition [simp]: "op_dfun_reset f k = f(k:=dflt)"  
  definition [simp]: "op_dfun_clear (f::'b⇒'a) = (λ_::'b. dflt)"  
  
  sepref_register 
    op_dfun_empty :: "('k,'v) i_dfun"
    op_dfun_lookup :: "('k,'v) i_dfun ⇒ 'k ⇒ 'v"
    op_dfun_upd :: "('k,'v) i_dfun ⇒ 'k ⇒ 'v ⇒ ('k,'v) i_dfun"
    op_dfun_reset :: "('k,'v) i_dfun ⇒ 'k ⇒ ('k,'v) i_dfun"
    op_dfun_clear :: "('k,'v) i_dfun ⇒ ('k,'v) i_dfun"
      
end

end
