text ‹Clause-IDs are limited to max_size-1, such that we can map them with a standard array-map›


definition "cid_rel ≡ b_rel size_rel (λn. n<max_size-1)"

lemma in_cid_rel_boundsD[sepref_bounds_dest]: 
  "(w, n) ∈ cid_rel ⟹ n < max_size-1"
  unfolding cid_rel_def
  by sepref_bounds

definition mk_cid :: "nat ⇒ nat" where [simp]: "mk_cid n = n"  
definition dest_cid :: "nat ⇒ nat" where [simp]: "dest_cid n = n"

  unfolding mk_cid_def cid_rel_def
