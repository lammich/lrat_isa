  lemma FOREACHcd_refine_stronger[refine]:
    assumes [simp]: "finite s ⟹ finite s'"
    assumes S: "(s',s)∈⟨S⟩set_rel"
    assumes SV: "single_valued S"
    assumes R0: "(σ',σ)∈R"
    assumes C: "⋀σ' σ. (σ',σ)∈R ⟹ (c' σ', c σ)∈bool_rel"
    assumes F: "⋀x' x σ' σ. ⟦(x', x) ∈ S; (σ', σ) ∈ R; x∈s⟧
       ⟹ f' x' σ' ≤ ⇓ R (f x σ)"
    shows "FOREACHcd s' c' f' σ' ≤ ⇓R (FOREACHcdi I s c f σ)"
  proof -
    have [refine_dref_RELATES]: "RELATES S" by (simp add: RELATES_def)
  
    from SV obtain α I where [simp]: "S=br α I" by (rule single_valued_as_brE)
    with S have [simp]: "s=α`s'" and [simp]: "∀x∈s'. I x" 
      by (auto simp: br_set_rel_alt)
    
    show ?thesis
      unfolding FOREACHcd_def FOREACHcdi_def
      
      find_in_thms nfoldli in refine
      
      find_theorems nfoldli list_rel
      
      
      apply (refine_rcg nfoldli_refine[where S="b_rel S (λx. x∈s)"])
      
      apply refine_dref_type
      subgoal by simp
      subgoal
        apply (auto simp: pw_le_iff refine_pw_simps)
        using S
        apply (rule_tac x="map α x" in exI)
        apply (auto simp: map_in_list_rel_conv)
        done
      subgoal 
        by (simp add: b_rel_list_rel_conv)
      subgoal using R0 by auto
      subgoal using C by auto  
      subgoal using F by force 
      done
  qed

  lemma unborrow'_rule[sepref_comb_rules]:
    assumes FRAME: "Γ ⊢ hn_ctxt R src srci ** hn_invalid R dst dsti ** F"
    assumes EQ: "src = dst"
    assumes EQI: "vassn_tag Γ ⟹ CP_cond (srci = dsti)"
    shows "hn_refine Γ (Mreturn ()) (hn_invalid R src srci ** hn_ctxt R dst dsti ** F) unit_assn (λ_. True) (RETURN$(unborrow'$src$dst))"
    apply (rule hn_refine_vassn_tagI)
    apply (rule hn_refine_cons_pre[OF FRAME])
    apply (rule hn_refineI)
    using EQ EQI
    unfolding unborrow'_def CP_defs
    apply (simp add: refine_pw_simps)
    unfolding hn_ctxt_def invalid_assn_def pure_def
    apply vcg'

  subsubsection ‹Grow and Ensure-Capacity›

  
  definition "wo_grow xinit l xs nsz ≡ doN {
    ASSERT (length xs ≤ nsz);
    ASSERT (l = length xs);
    
    xs ← mop_to_eo_conv xs;
    
    let ys = op_woarray_replicate xinit nsz; ⌦‹Assertion annotation to be inserted only upon refinement›
    ys ← mop_to_eo_conv ys;
    
    (_,xs,ys)←WHILEIT (λ(i,xs',ys'). i≤l ∧ xs' = replicate i None @ drop i xs ∧ ys' = take i xs @ replicate (nsz-i) (Some xinit)) 
      (λ(i,xs,ys). i<l) 
      (λ(i,xs,ys). doN {
        (x,xs) ← mop_eo_extract xs i;
        (y,ys) ← mop_eo_extract ys i;  
        mop_free_dflt xinit y;
        
        ys ← mop_eo_set ys i x;
        ASSERT (i+1≤l);
        RETURN (i+1,xs,ys)
      }) (0,xs,ys);
  
    mop_eoarray_free_empty xs;
    
    ys ← mop_to_wo_conv ys;
      
    RETURN ys
  }"  
    
  
  definition "wo_ensure_capacity xinit l xs i ≡ doN {
    if i<l then
      RETURN (l,xs)
    else if l ≥ max_size_incl div 2 then doN {
      let l' = i+1;
      xs ← wo_grow xinit l xs l';
      RETURN (l',xs)
    } else doN {
      ASSERT (2*l < max_size);
      let l' = max (2*l) (i+1);
      xs ← wo_grow xinit l xs l';
      RETURN (l',xs)
    }
  }"
  
  
  lemma wo_grow_rule[refine_vcg]: "⟦l=length xs; length xs ≤ nsz⟧ ⟹ wo_grow xinit l xs nsz ≤ SPEC (λys. ys = xs @ replicate (nsz - length xs) xinit)"
    unfolding wo_grow_def mop_to_eo_conv_def
    apply (refine_vcg WHILEIT_rule[where R="measure (λ(i,_,_). l-i)"])
    apply (auto simp: nth_append replicate_append_same upd_conv_take_nth_drop take_Suc_conv_app_nth)
    done
  
  lemma wo_ensure_capacity_rule[refine_vcg]: "⟦ l = length xs ⟧ ⟹ wo_ensure_capacity xinit l xs i 
    ≤ SPEC (λ(l',ys). length xs ≤ length ys ∧ i<length ys ∧l'=length ys ∧ (∀j∈{0..<length xs}. ys!j=xs!j) ∧ (∀j∈{length xs..<length ys}. ys!j=xinit))"
    unfolding wo_ensure_capacity_def
    apply refine_vcg
    by (auto simp: nth_append)
